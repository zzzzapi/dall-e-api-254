API_KEY = "your_4Z_api_key"
API_URL = "https://api.zzzzapi.com/v1/chat/completions"

    "Authorization": f"Bearer {API_KEY}",
    "Content-Type": "application/json"
}

data = {
    "model": "gpt-4o",
    "messages": [
        {"role": "system", "content": "你是一个专业AI助手"},
        {"role": "user", "content": "请介绍OpenAI API的主要功能"}
    ],
    "temperature": 0.7,
    "max_tokens": 1000
}

response = requests.post(API_URL, headers=headers, json=data)
print(response.json())
